    # Scala 的集合介绍：

    序列 Seq、集 Set、映射Map，几乎所有的集合类，Scala都提供了可变和不可变的版本。

    Scala 不可变集合：就是指该集合对象不可修改，每次修改就会返回一个新对象，而不会对原对象进行修改。类似于java中的String对象

    Scala 可变集合：就是这个集合可以直接对原对象进行修改，而不会返回新的对象。类似于java中StringBuilder 对象

     val newArr1 = arr1 :+ 15
     println(arr1) //ArrayBuffer()

     val newArr2 = arr1 += 19

     arr1 -= 13 //删除元素为13的值
     println(arr1) //ArrayBuffer(15, 59, 76, 77, 19, 36)
 //（4）修改元素

    列表(List|ListBuffer):

    (1)不可变列表

    定义不可变列表方式:val list: List[Int] = List(1,2,3,4,3)

    def main(args: Array[String]): Unit = {
    // 1. 创建一个List
    val list1 = List(23, 65, 87)
    println(list1) //List(23, 65, 87)

    // 2. 访问和遍历元素
    println(list1(1)) //65
//    list1(1) = 12
    list1.foreach(println) //23 65 87

    // 3. 添加元素
    val list2 = 10 +: list1 //往前添加一个元素
    val list3 = list1 :+ 23 //往后添加一个元素
    println(list1) //List(23, 65, 87)
    println(list2) //List(10, 23, 65, 87)
    println(list3) //List(23, 65, 87, 23)

    println("==================")

    val list4 = list2.::(51)
    println(list4) //List(51, 10, 23, 65, 87)

    val list5 = Nil.::(13) //Nil表示空列表
    println(list5) //List(13)

    val list6 = 73 :: 32 :: Nil
    val list7 = 17 :: 28 :: 59 :: 16 :: Nil
    println(list7) //List(17, 28, 59, 16)

    // 4. 合并列表
    val list8 = 55 :: list6 :: list7
    println(list8) //List(55, List(73, 32), 17, 28, 59, 16)

    val list9 = list6 ::: list7
    println(list9) //List(73, 32, 17, 28, 59, 16)

    val list10 = list6 ++ list7
    println(list10) //List(73, 32, 17, 28, 59, 16)

    // 5. 列表的特殊用法
    val List(first, second, _*) = List(23, 15, 9, 78)
    println(s"first: $first, second: $second")

    val fir :: sec :: rest = List(23, 15 , 9, 78)
    println(s"first: $fir, second: $sec, rest: $rest")

  }

    (2)可变列表

    定义可变列表方式:val buffer = ListBuffer(1,2,3,4)

    def main(args: Array[String]): Unit = {
    // 1. 创建可变列表
    val list1: ListBuffer[Int] = new ListBuffer[Int]()
    val list2 = ListBuffer(12, 53, 75)

    println(list1) //ListBuffer()
    println(list2) //ListBuffer(12, 53, 75)

    println("==============")

    // 2. 添加元素
    list1.append(15, 62)
    list2.prepend(20)

    list1.insert(1, 19, 22) //指定在索引位置1处添加两个元素

    println(list1) //ListBuffer(15, 19, 22, 62)
    println(list2) //ListBuffer(20, 12, 53, 75)

    println("==============")

    31 +=: 96 +=: list1 += 25 += 11
    println(list1) //ListBuffer(31, 96, 15, 19, 22, 62, 25, 11)

    println("==============")
    // 3. 合并list
    val list3 = list1 ++ list2
    println(list1) //ListBuffer(31, 96, 15, 19, 22, 62, 25, 11)
    println(list2) //ListBuffer(20, 12, 53, 75)
    println(list3) //ListBuffer(31, 96, 15, 19, 22, 62, 25, 11, 20, 12, 53, 75)
    println("==============")

    list1 ++=: list2
    println(list1) //ListBuffer(31, 96, 15, 19, 22, 62, 25, 11)
    println(list2) //ListBuffer(31, 96, 15, 19, 22, 62, 25, 11, 20, 12, 53, 75)

    println("==============")

    // 4. 修改元素
    list2(3) = 30
    list2.update(0, 89)
    println(list2) //ListBuffer(89, 96, 15, 30, 22, 62, 25, 11, 20, 12, 53, 75)

    // 5. 删除元素
    list2.remove(2)
    list2 -= 25
    println(list2) //ListBuffer(89, 96, 30, 22, 62, 11, 20, 12, 53, 75)
  }

    队列(Queue|mutable.Queue):

    定义不可变队列方式:val que = Queue[String]()

    定义可变队列方式:val que = new mutable.Queue[String]()

    def main(args: Array[String]): Unit = {
    // 创建一个可变队列
    val queue: mutable.Queue[String] = new mutable.Queue[String]()

    queue.enqueue("a", "b", "c")

    println(queue) //Queue(a, b, c)
    println(queue.dequeue()) // a
    println(queue) //Queue(b, c)
    println(queue.dequeue()) // b
    println(queue) //Queue(c)

    queue.enqueue("d", "e")

    println(queue) //Queue(c, d, e)
    println(queue.dequeue()) // c
    println(queue) //Queue(d, e)

    println("==========================")

    // 不可变队列
    val queue2: Queue[String] = Queue[String]("a", "b", "c")
    val queue3 = queue2.enqueue("d")
    println(queue2) //Queue(a, b, c)
    println(queue2.dequeue) //(a,Queue(b, c))
    println(queue2) //Queue(a, b, c)
    println(queue3) //Queue(a, b, c, d)
  }

    # 2、集Set

    Set集(Set|mutable.Set):

    定义不可变Set方式:val set = Set(1,2,3,4,5,6)

    def main(args: Array[String]): Unit = {
    // 1. 创建set
    val set1 = Set(13, 23, 53, 12, 13, 23, 78)
    println(set1) //Set(78, 53, 13, 12, 23)

    println("==================")

    // 2. 添加元素
    val set2 = set1 + 129
    println(set1) //Set(78, 53, 13, 12, 23)
    println(set2) //Set(78, 53, 13, 129, 12, 23)
    println("==================")

    // 3. 合并set
    val set3 = Set(19, 13, 23, 53, 67, 99)
    val set4 = set2 ++ set3
    println(set2) //Set(78, 53, 13, 129, 12, 23)
    println(set3) //Set(53, 13, 67, 99, 23, 19)
    println(set4) //Set(78, 53, 13, 129, 12, 67, 99, 23, 19)

    // 4. 删除元素
    val set5 = set3 - 13
    println(set3) //Set(53, 13, 67, 99, 23, 19) set5变化不影响set3集合
    println(set5) //Set(53, 67, 99, 23, 19)

    // 5. 遍历元素
    for(x<-set5){
      println(x)
    }
  }

    # 3、映射Map

    Map集(Map|mutable.Set):

    定义不可变Map方式:val map = Map( "a"->1, "b"->2, "c"->3 )

    def main(args: Array[String]): Unit = {
    // 1. 创建map
    val map1: Map[String, Int] = Map("a" -> 13, "b" -> 25, "hello" -> 3)
    println(map1) //Map(a -> 13, b -> 25, hello -> 3)
    println(map1.getClass) //class scala.collection.immutable.Map$Map3
    println("==================================")
    // 2. 遍历元素
    map1.foreach(println)
    map1.foreach( (kv: (String, Int)) => println(kv) )

    println("==================================")

    // 3. 取map中所有的key 或者 value
    for (key <- map1.keys){
      println(s"$key ---> ${map1.get(key)}")
    }
    println("==================================")
    // 4. 访问某一个key的value
    println("a: " + map1.get("a").get) //a: 13
    println("a: " + map1("a")) //a: 13
    println("c: " + map1.get("c")) //c: None
    println(map1.get("c").getOrElse("没有存在该值")) //没有存在该值
    println(map1.getOrElse("c", "没有存在该值")) //没有存在该值
    println("c: " + map1.getOrElse("c", 0)) //c: 0
  }

    def main(args: Array[String]): Unit = {
    // 1. 创建map
    val map1: mutable.Map[String, Int] = mutable.Map("a" -> 13, "b" -> 25, "hello" -> 3)
    println(map1) //Map(b -> 25, a -> 13, hello -> 3)
    println(map1.getClass) //class scala.collection.mutable.HashMap

    println("==========================")

    // 2. 添加元素
    map1.put("c", 5)
    map1.put("d", 9)
    println(map1) //Map(b -> 25, d -> 9, a -> 13, c -> 5, hello -> 3)

    map1 += (("e", 7))
    println(map1) //Map(e -> 7, b -> 25, d -> 9, a -> 13, c -> 5, hello -> 3)

    println("==========================")

    // 3. 删除元素
    println(map1("c")) //5
    map1.remove("c")
    println(map1.getOrElse("c", 0)) //0

    map1 -= "d"
    println(map1) //Map(e -> 7, b -> 25, a -> 13, hello -> 3)

    println("==========================")

    // 4. 修改元素
    map1.update("c", 5)
    map1.update("e", 10)
    println(map1) //Map(e -> 10, b -> 25, a -> 13, c -> 5, hello -> 3)

    println("==========================")

    // 5. 合并两个Map
    val map2: Map[String, Int] = Map("aaa" -> 11, "b" -> 29, "hello" -> 5)
//    map1 ++= map2
    println(map1) //Map(e -> 10, b -> 25, a -> 13, c -> 5, hello -> 3)
    println(map2) //Map(aaa -> 11, b -> 29, hello -> 5)

    println("==========================")
    val map3: Map[String, Int] = map2 ++ map1
    val map4 = map1 ++ map2 //等价于map1 ++= map2
    println(map1) //Map(e -> 10, b -> 25, a -> 13, c -> 5, hello -> 3)
    println(map2) //Map(aaa -> 11, b -> 29, hello -> 5)
    println(map3) //Map(e -> 10, a -> 13, b -> 25, c -> 5, hello -> 3, aaa -> 11)
    println(map4) //Map(e -> 10, aaa -> 11, b -> 29, a -> 13, c -> 5, hello -> 5)
  }

    # 4、元组Tuple(索引从1开始)

    定义元组的方式:val tuple: (Int, String, Boolean) = (40,"lisi",true)

    def main(args: Array[String]): Unit = {
    // 1. 创建元组
    val tuple: (String, Int, Char, Boolean) = ("hello", 100, 'a', true)
    println(tuple)

    // 2. 访问数据
    println(tuple._1)
    println(tuple._2)
    println(tuple._3)
    println(tuple._4)
    
    // 通过索引访问数据
    println(tuple.productElement(1))

    println("====================")
    // 3. 遍历元组数据
    for (elem <- tuple.productIterator)
      println(elem)

    // 4. 嵌套元组
    val mulTuple = (12, 0.3, "hello", (23, "scala"), 29)
    println(mulTuple._4._2)
  }

    # Scala集合操作：

    # Scala集合的基础函数操作：

      def main(args: Array[String]): Unit = {
      val list1: List[Int] = List(1, 2, 3, 4, 5, 6, 7)
      val list2: List[Int] = List(4, 5, 6, 7, 8, 9, 10)
      //（1）获取集合长度
      println(list1.length)
      //（2）获取集合大小,等同于 length
      println(list1.size)
      //（3）循环遍历
      list1.foreach(println)
      println("=======================================")
      //（4）迭代器
      for (elem <- list1.iterator) {
        println(elem)
      }
      println("=======================================")
      //（5）生成字符串
      println(list1.mkString(","))
      //（6）是否包含
      println(list1.contains(3))
      //（7）获取集合的头
      println(list1.head)
      //（8）获取集合的尾（不是头的就是尾部数据）
      println(list1.tail)
      //（9）集合最后一个数据
      println(list1.last)
      //（10）获取集合的头（不是尾的就是头部数据）
      println(list1.init)
      //（11）反转
      println(list1.reverse)
      //（12）取前（后）n 个元素
      println(list1.take(3))
      println(list1.takeRight(3))
      //（13）去掉前（后）n 个元素
      println(list1.drop(3))
      println(list1.dropRight(3))
      //（14）并集
      println(list1.union(list2))
      //（15）交集
      println(list1.intersect(list2))
      //（16）差集(list1独有的元素，谁点就谁独有)
      println(list1.diff(list2)) //List(1, 2, 3)
      //（17）拉链 注:如果两个集合的元素个数不相等，那么会将同等数量的数据进行拉链，多余的数据省略不用
      println(list1.zip(list2)) //List((1,4), (2,5), (3,6), (4,7), (5,8), (6,9), (7,10))
      //（18）滑窗,sliding第一个参数表示取几个值,第二个参数表示从哪个索引开始取值（包含，比如从第4位）
      list1.sliding(3, 4).foreach(println)  //List(5, 6, 7)
      //（19）求和
      println(list1.sum)
      //（20）求乘积
      println(list1.product)
      //（21）最大值
      println(list1.max)
      //（22）最小值
      println(list1.min)
      //（23）排序
      // 按照元素大小排序(默认升序)
      println(list1.sortBy(x => x))
      // 按照元素的绝对值大小排序(默认升序)
      println(list1.sortBy(x => x.abs))
      // 按元素大小升序排序
      println(list1.sortWith((x, y) => x < y))
      // 按元素大小降序排序
      println(list1.sortWith((x, y) => x > y))
    }
